=== Drawing a circle
[livecode,ex-1]
++++
function love.draw()
    love.graphics.setColor( 255, 0, 0, 255 )
    love.graphics.circle( "fill", 200, 200, 100 )
end
++++

=== Adding a tad of complexity (and rects)
[livecode,ex-2]
++++
function rect()
    return 200, 200, 100, 100
end

function love.draw()
    love.graphics.rectangle( "fill", rect() )
end
++++
